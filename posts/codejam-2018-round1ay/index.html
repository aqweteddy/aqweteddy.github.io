CodeJam 2018 Round1A

A. WaffleChoppers

題目描述

  • URL

  • Input:

    • R: MAX_ROW, C: MAX_COL, H: 水平方向切幾刀 V: 垂直方向切幾刀
    • 一個二維矩陣 [R*C]：每格為 1 or 0

  • 在水平方向切 H 刀、垂直方向切 V 刀的情況下，問可否達到每個區域都有相同數量的 1

  • Output: 可否達成目標

解題思路

  1. 先判斷出特殊解：
    • 矩陣全為 0
  2. 因為每格要相同數量，所以整張圖1的數量應要整除於 H+1 和 V+1
  3. 計算每個 Row 的 1 的數量，記錄在一維陣列 r_tab
  4. 計算每個 Col 的 1 的數量，記錄在一維陣列 c_tab
  5. 因為每格要相同數量，所以直著切要可以平分成 H+1 等分
    • 紀錄切點位置
    • 根據 r_tab 切分
    • 如果無粉平分，則無法達成目標
  6. 因為每格要相同數量，所以橫著切要可以平分成 V+1 等分
    • 紀錄切點位置
    • 根據 c_tab 切分
    • 如果無粉平分，則無法達成目標
  7. 根據切點檢查每格是否是相同數量
    • 若是，達成目標
    • 否則，無法達成
  8. 演算法效率：O(R * C)，因為 Step 7 每個區域都要掃過一次

Code

// AC

#include <bits/stdc++.h>

using namespace std;

int maze[101][101];
int R, C, H, V; // row, col, horizontal, vertical
int cnt_all; // count for the number of chocolate chips.

bool bound_count(int min_h, int min_v, int max_h, int max_v, int per)
{
    int cnt = 0;
    for (int i = min_h; i <= max_h; ++i)
        for (int j = min_v; j <= max_v; ++j)
            if (maze[i][j] == 1)
                ++cnt;
    return cnt == per;
}

void solve()
{
    if (cnt_all % ((H + 1) * (V + 1)) != 0) {
        cout << "IMPOSSIBLE\n";
        return;
    }
    if (cnt_all == 0) {
        cout << "POSSIBLE\n";
        return;
    }
    int r_tab[101], c_tab[101];

    // generate table
    memset(r_tab, 0, sizeof r_tab);
    memset(c_tab, 0, sizeof c_tab);

    for (int i = 0; i < R; ++i) {
        for (int j = 0; j < C; ++j)
            r_tab[i] += maze[i][j];
    }
    for (int i = 0; i < C; ++i) {
        for (int j = 0; j < R; ++j)
            c_tab[i] += maze[j][i];
    }
    // calculate
    vector<int> r_cut, c_cut;
    int r_per = cnt_all / (H + 1),
        c_per = cnt_all / (V + 1);

    int now = 0;
    for (int i = 0; i < R; ++i) {
        now += r_tab[i];
        if (now == r_per) {
            now = 0;
            r_cut.push_back(i);
        } else if (now > r_per) {
            cout << "IMPOSSIBLE\n";
            return;
        }
    }
    now = 0;

    for (int i = 0; i < C; ++i) {
        now += c_tab[i];
        if (now == c_per) {
            now = 0;
            c_cut.push_back(i);
        } else if (now > c_per) {
            cout << "IMPOSSIBLE\n";
            return;
        }
    }

    int per = r_per / (V + 1);
    int r_pre = 0;
    int c_pre = 0;

    for (int i = 0; i < r_cut.size(); ++i) {
        for (int j = 0; j < c_cut.size(); ++j) {
            if (!bound_count(r_pre, c_pre, r_cut[i], c_cut[j], per)) {
                cout << "IMPOSSIBLE\n";
                return;
            }
            c_pre = c_cut[j] + 1;
        }
        c_pre = 0;
        r_pre = r_cut[i] + 1;
    }
    cout << "POSSIBLE\n";
}

int main()
{
    int ncase;

    cin >> ncase;
    for (int n = 0; n < ncase; ++n) {
        string in;
        memset(maze, 0, sizeof maze);
        cin >> R >> C >> H >> V;
        cnt_all = 0;
        for (int i = 0; i < R; ++i) {
            cin >> in;
            for (int j = 0; j < in.size(); ++j) {
                if (in[j] == '@')
                    maze[i][j] = 1, ++cnt_all;
                else
                    maze[i][j] = 0;
            }
        }
        cout << "Case #" << n + 1 << ": ";
        solve();
    }
    return 0;
}

B. BitParty

題目描述

  • URL

  • 有 C 台機器要平行處理 B 個物品，一次可啟用 R 台，其中每台機器最多可處理 M[i] 個物品，需要花費 (S[i] * N) + P[i], N <= M[i] 的時間，問：處理完 B 個物品，最少花費多少時間？

解題思路

  • 二分搜時間 T(t)
    • 若在 t 內可做到 mid = l
    • 若在 t 內不可做到 mid = r
  • 如何算「第 i 台機器在 t 秒內所能處理的最大物品數(N[i])」?
    1. 因為 t = (S[i] * N) + P[i]
    2. 所以 N = (t - P[i]) / S[i]，其中 N <= M[i]
    3. 所以 N[i] = min(M[i], (t - P[i]) / S[i])
    4. 考慮 t < P[i] 時值為負，N[i] = max(0, min(M[i], (t - P[i]) / S[i]))
  • 遞增排序 Array N，持續將物品放入機器中，直到：
    • 已放入物品數量 >= B –> return True
    • 已放入物品數量 >= min(R, C) –> return False

注意

  1. long long
  2. 二分搜的 Initial Value
  3. INF 值:
    • int: INT_MAX or 0x3f3f3f3f
    • long long: __LONG_LONG_MAX__
  4. 0ll: 將 int 轉為 long long

Code

#include <bits/stdc++.h>
// AC
using namespace std;
typedef long long ll;
const ll INF = __LONG_LONG_MAX__;
ll R, B, C;

struct Cashier
{
    ll m, s, p;
    Cashier(ll a, ll b, ll c)
    {
        m = a, s = b, p = c;
    }
};
vector<Cashier> inp;

bool cmp(const ll&a, const ll &b)
{
    return a > b;
}

bool check(ll t)
{
    vector<ll> bit;
    for(const Cashier& c : inp){
        ll tmp = max(0ll, min((t - c.p) / c.s, c.m));
        bit.push_back(tmp);
    }
    sort(bit.begin(), bit.end(), cmp);
    ll now = 0;
    for(ll i=0; i<bit.size(); ++i){
        now += bit[i];
        if(now >= B) return 1;
        if(i + 1 == R) break;
    }
    return 0;
}

void solve()
{
    ll l = 0, r = INF, ans;
    for(int i=0; i<100; ++i){
        ll mid = (l + r) >> 1;
        if( check(mid) ) {
            r = mid;
        }
        else l = mid;
    }
    cout << r << '\n';
}

int main()
{
    int nc;
    cin >> nc;
    for(int n=1; n<=nc; ++n){
        inp.clear();
        cin >> R >> B >> C;
        for(int i=0; i<C; ++i){
            ll a, b, c;
            cin >> a >> b >> c;
            inp.push_back(Cashier(a, b, c));
        }
        cout <<"Case #" << n << ": ";
        solve();
    }
    return 0;
}

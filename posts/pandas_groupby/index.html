Pandas groupby 筆記

一些準備

import numpy as np
import pandas as pd

df = pd.DataFrame({
    'key': ['a', 'b', 'c', 'a', 'b', 'c'],
    'd1': np.arange(6),
    'd2': np.random.randint(0, 100, 6)
})

基本 groupby

df.groupby('key') # 按 key 分類，return DataFrameGroupBy Object
df.groupby('key').sum() # return 每一 col 的 sum
df.groupby('key')[['d1', 'd2']].median() # return col a, b 的 median

aggregate

  1. 對整體操作
  • 傳 list
df.groupby('key').aggregate([np.min, np.median, np.max])

# output
      d1               d2            
keys  amin median amax amin median amax                                  
a      0    1.5    3    2    9.0   16
b      1    2.5    4    7   17.5   28
c      2    3.5    5   16   50.5   85
  1. 分別對各項操作
  • 傳 dict
df.groupby('key').aggregate({'d1':np.min, 'd2':np.max}) # d1: a類中max, d2: a類中min

# output
key d1 	d2 		
a   0 	16
b   1 	28
c   2 	85

filter

  • 基於群組特性丟棄資料
print(df)
# output
  key  d1  d2
0   a   0  16
1   b   1   7
2   c   2  16
3   a   3   2
4   b   4  28
5   c   5  85

print(df.groupby('key').std())
# output
key    d1         d2             
a    2.12132   9.899495
b    2.12132  14.849242
c    2.12132  48.790368

def f(x):
    return x['d2'].std() > 13
print(df.groupby('key').filter(f)) # 透過 x['d2'] std > 13 把 group a 過濾掉
# output
   key  d1  d2
1   b   1   7
2   c   2  16
4   b   4  28
5   c   5  85

transform

df.groupby('key').transform(lambda x: x - x.mean())

# output

	  d1 	  d2
0 	-1.5 	7.0
1 	-1.5 	-10.5
2 	-1.5 	-34.5
3 	1.5 	-7.0
4 	1.5 	10.5
5 	1.5 	34.5

apply

def d1_sum(x):
    x['d3'] = x['d1'].sum()
    return x

print(df.groupby('key').apply(d1_sum)) # d3: group a group b group c 的和

# output
  key  d1  d2  d3
0   a   0  16   3
1   b   1   7   5
2   c   2  16   7
3   a   3   2   3
4   b   4  28   5
5   c   5  85   7

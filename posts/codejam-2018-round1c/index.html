CodeJam 2018 Round1C

A. A Whole New Word

題目描述

  • URL
  • 給定一個字元矩陣 \(N \times L\)，每個 column 選一個字母組成一個新單字，但不和 row 成的單字重複，輸出任意一個單字，若無輸出“-”

解題思路

  • 暴力 + 剪枝
  • 紀錄每個 column
  • 遞迴拼出所有可能性
  • 剪枝：
    • 開一個陣列紀錄 column[k] 下 A-Z 是否使用過
    • 若沒有：遞迴，並將 flag 設為 1
    • 若有：跳過
  • 如此一來，每從一個 Column 決定一個字元最多只會呼叫 26 次遞迴

Code

#include <bits/stdc++.h>

using namespace std;

set<string> exist;
vector<string> dic;
int L, N;
bool fl;

void generate_word(string str, int k)
{
    if(fl) return;
    if(k == L){
        if(find(exist.begin(), exist.end(), str) == exist.end()){
            cout << str << '\n';
            fl = 1;
        }
        return;
    }
    bool used[30];
    for(int i=0; i<30; ++i)
        used[i] = 0;
    for(int i=0 ;i<N; ++i){
       if(!used[dic[k][i] - 'A']){
            generate_word(str + dic[k][i], k + 1);
            used[dic[k][i] - 'A'] = 1;
       }
    }
}

void solve()
{
    generate_word("", 0);
    if(!fl) cout << "-\n";
}

int main()
{
    int ncase;

    cin >> ncase;
    for (int n = 0; n < ncase; ++n) {
        cin >> N >> L;
        string in;
        dic.clear();
        exist.clear();
        dic.resize(L);
        fl = 0;
        for(int i=0; i<N; ++i){
            cin >> in;
            exist.insert(in);
            for(int j=0; j<in.size(); ++j){
                dic[j] += in[j];
            }
        }
        cout << "Case #" << n + 1 << ": ";
        solve();
    }
    return 0;
}

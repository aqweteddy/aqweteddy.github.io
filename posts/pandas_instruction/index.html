缺失資料

  • NaN(float), None
  • 所有計算遇到 NaN 時，答案為 NaN

處理 null

pla.isnull() # 列出表格，null 為 True，其餘為false
pla.isnull().any() # 列出任何包含 null 的 column 名
pla.notnull() # 列出表格，null 為 False，其餘為 True

pla.dropna() # 傳回一個過濾後的資料(移除任何出現null的那個row)
pla.dropna(how='all') # 傳回一個過濾後的資料(移除全部都是 null的那個row)
pla.dropna(axis='columns') # 移除任何出現null的那個 column

pla.fillna(k) # 將 na 填入 k 值
pla.fillna(method='ffill') # 使用前一個值填入
pla.fillna(method='bfill') # 使用下一個值填入
pla.fillna(method='bfill', axis='columns') # 使用下一個值填入(對 column)

df.query()

pla.query('number > 1 and mass < 7') # 符合條件 pla['number'] > 1 and pla['mass'] < 7

a = 10
pla.query('number > 1 and mass < @a') # 引入變數 a

# output

df.eval()

  • df.eval(運算式)
pla.eval('mass / distance') # pla['mass'] / pla['distance']
pla.eval('s = mass / distance', inplace=True) # 多一欄 pla[s] = pla['mass'] / pla['distance']
pla.eval('s = mass / distance') # 多一欄 pla[s] = pla['mass'] / pla['distance']，但不會更動原始 pla

a = 1
pla.eval('s = mass / distance + @a') # 引入變數 a

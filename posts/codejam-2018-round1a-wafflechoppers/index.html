2018-Round1A WaffleChoppers

題目描述

  • URL

  • Input:

    • R: MAX_ROW, C: MAX_COL, H: 水平方向切幾刀 V: 垂直方向切幾刀
    • 一個二維矩陣 [R*C]：每格為 1 or 0

  • 在水平方向切 H 刀、垂直方向切 V 刀的情況下，問可否達到每個區域都有相同數量的 1

  • Output: 可否達成目標

解題思路

  1. 先判斷出特殊解：
    • 矩陣全為 0
  2. 因為每格要相同數量，所以整張圖1的數量應要整除於 H+1 和 V+1
  3. 計算每個 Row 的 1 的數量，記錄在一維陣列 r_tab
  4. 計算每個 Col 的 1 的數量，記錄在一維陣列 c_tab
  5. 因為每格要相同數量，所以直著切要可以平分成 H+1 等分
    • 紀錄切點位置
    • 根據 r_tab 切分
    • 如果無粉平分，則無法達成目標
  6. 因為每格要相同數量，所以橫著切要可以平分成 V+1 等分
    • 紀錄切點位置
    • 根據 c_tab 切分
    • 如果無粉平分，則無法達成目標
  7. 根據切點檢查每格是否是相同數量
    • 若是，達成目標
    • 否則，無法達成

Code

// AC

#include <bits/stdc++.h>

using namespace std;

int maze[101][101];
int R, C, H, V; // row, col, horizontal, vertical
int cnt_all; // count for the number of chocolate chips.

bool bound_count(int min_h, int min_v, int max_h, int max_v, int per)
{
    int cnt = 0;
    for (int i = min_h; i <= max_h; ++i)
        for (int j = min_v; j <= max_v; ++j)
            if (maze[i][j] == 1)
                ++cnt;
    return cnt == per;
}

void solve()
{
    if (cnt_all % ((H + 1) * (V + 1)) != 0) {
        cout << "IMPOSSIBLE\n";
        return;
    }
    if (cnt_all == 0) {
        cout << "POSSIBLE\n";
        return;
    }
    int r_tab[101], c_tab[101];

    // generate table
    memset(r_tab, 0, sizeof r_tab);
    memset(c_tab, 0, sizeof c_tab);

    for (int i = 0; i < R; ++i) {
        for (int j = 0; j < C; ++j)
            r_tab[i] += maze[i][j];
    }
    for (int i = 0; i < C; ++i) {
        for (int j = 0; j < R; ++j)
            c_tab[i] += maze[j][i];
    }
    // calculate
    vector<int> r_cut, c_cut;
    int r_per = cnt_all / (H + 1),
        c_per = cnt_all / (V + 1);

    int now = 0;
    for (int i = 0; i < R; ++i) {
        now += r_tab[i];
        if (now == r_per) {
            now = 0;
            r_cut.push_back(i);
        } else if (now > r_per) {
            cout << "IMPOSSIBLE\n";
            return;
        }
    }
    now = 0;

    for (int i = 0; i < C; ++i) {
        now += c_tab[i];
        if (now == c_per) {
            now = 0;
            c_cut.push_back(i);
        } else if (now > c_per) {
            cout << "IMPOSSIBLE\n";
            return;
        }
    }

    int per = r_per / (V + 1);
    int r_pre = 0;
    int c_pre = 0;

    for (int i = 0; i < r_cut.size(); ++i) {
        for (int j = 0; j < c_cut.size(); ++j) {
            if (!bound_count(r_pre, c_pre, r_cut[i], c_cut[j], per)) {
                cout << "IMPOSSIBLE\n";
                return;
            }
            c_pre = c_cut[j] + 1;
        }
        c_pre = 0;
        r_pre = r_cut[i] + 1;
    }
    cout << "POSSIBLE\n";
}

int main()
{
    int ncase;

    cin >> ncase;
    for (int n = 0; n < ncase; ++n) {
        string in;
        memset(maze, 0, sizeof maze);
        cin >> R >> C >> H >> V;
        cnt_all = 0;
        for (int i = 0; i < R; ++i) {
            cin >> in;
            for (int j = 0; j < in.size(); ++j) {
                if (in[j] == '@')
                    maze[i][j] = 1, ++cnt_all;
                else
                    maze[i][j] = 0;
            }
        }
        cout << "Case #" << n + 1 << ": ";
        solve();
    }
    return 0;
}
